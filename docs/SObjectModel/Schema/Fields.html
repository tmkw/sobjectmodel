def find_by(name: nil, label: nil)
  return nil if name.nil? && label.nil?

  attr_name = name.nil? ? :label : :name
  val       = name || label

  find do |field|
    attr_val = field.__send__(attr_name.to_sym)
    attr_val == val.to_s
  end
end
